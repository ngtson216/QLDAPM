#include<iostream>

using namespace std;

int main() {
    int N;
    cin >> N;
    char a = '1';
    char index = '-1';
if(N >=0){
    string s;
    cin >> s ;
    for (int i = 0; i < N; i++) {
        if (s[i] == a) {
            index = i;
            break;
        }
    }
    if (index % 2 == 0)
        cout << "Takahashi";
    else
        cout << "Aoki";
}
}
